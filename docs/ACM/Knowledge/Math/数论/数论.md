### 1.1.6大区间素数筛选（acwing 196）

```cpp
/*
acwing 196
在闭区间[L, U]内,找出相邻的，距离最近的两个素数和距离最远的两个素数
1 <= L < U <= 2147483647, 区间长度不超过 1000000
*/

const int N = 1000010;

int primes[N], cnt;
bool st[N];

void init(int n)
{
    memset(st, 0, sizeof st);
    cnt = 0;
    for(int i = 2; i <= n; i ++)
    {
        if(!st[i]) primes[cnt ++] = i;
        for(int j = 0; primes[j] * i <= n; j ++)
        {
            st[primes[j] * i] = true;
            if(i % primes[j] == 0) break;
        }
    }
}

int main()
{
    int l, r;
    while(cin >> l >> r)
    {
        init(50000);

        memset(st, 0, sizeof st);
        for(int i = 0; i < cnt; i ++)
        {
            LL p = primes[i];
            for(LL j = max(2 * p, (l + p - 1) / p * p); j <= r; j += p)
                st[j - l] = true;
        }

        cnt = 0;
        for(int i = 0; i <= r - l; i ++)
            if(!st[i] && i + l >= 2)
                primes[cnt ++] = i + l;

        if(cnt < 2) puts("There are no adjacent primes.");
        else
        {
            int minp = 0, maxp = 0;
            for(int i = 0; i + 1 < cnt; i ++)
            {
                int d = primes[i + 1] - primes[i];
                if(d < primes[minp + 1] - primes[minp]) minp = i;
                if(d > primes[maxp + 1] - primes[maxp]) maxp = i;
            }

            printf("%d,%d are closest, %d,%d are most distant.\n",
                primes[minp], primes[minp + 1],
                primes[maxp], primes[maxp + 1]);
        }
    }

    return 0;
}
```

## 1.2约数

### 1.2.0前置知识

a.算术基本定理：

对于任意正整数N = $P1^{a1}*P2^{a2}*\cdots*Pn^{an}$

这里P1 < P2 < ··· < Pn均为质数，其中指数ai是正整数

b.求约数的个数：

S = (a1 + 1) * (a2 + 1) * ··· *(an + 1)

c.求所有约数的和（容斥原理）

S = $(P1^0+P1^1+\cdots+P1^{a1})*(P2^0+P2^1+\cdots+P2^{a2})*\cdots*(Pn^0+Pn^1+\cdots+Pn^{an})$

```cpp
            x /= i;
```

## 1.4欧拉函数

欧拉定理：若a与n互质，则 $a^{\varphi(n)}\equiv 1 \;(\bmod n)$

### 1.4.1欧拉函数 O($\sqrt{a}$)

定义：对于正整数n，欧拉函数是小于等于n的正整数中与n互质的数的数目，记作$\varphi(n)$,其中有$\varphi(1) = 1$。

公式：$\varphi(n) = n * (1 - \frac{1}{p1}) * (1 - \frac{1}{p2}) * \cdots * (1 - \frac{1}{pn})$

原理：容斥原理

```cpp
int get_eular(int n)
{
    int res = n;
    for(int i = 2; i <= n / i ; i ++)
        if(n % i == 0)
        {
            while(n % i == 0) n /= i;
            res =  res / i * (i - 1);
        }
    if(n > 1) res = res / n * (n - 1);
    
    return res;
}
```

### 1.4.2线性筛欧拉函数 O(n)

1.如果i为质数，则 phi[i] = i - 1

2.如果 i % primes[j] == 0, 则 phi[primes[j] * i] = primes[j] * phi[i]

3.如果 i % primes[j] != 0, 则 phi[primes[j] * i] = (primes[j] - 1) * phi[i]

```cpp
void get_eulers(int n)
{
    phi[1] = 1;
    for(int i = 2; i <= n; i ++)
    {
        if(!st[i])
        {
            primes[cnt ++] = i;
            phi[i] = i - 1;
        }
        
        for(int j = 0; primes[j] * i <= n; j ++)
        {
            st[primes[j] * i] = true;
            if(i % primes[j] == 0)
            {
                phi[primes[j] * i] = primes[j] * phi[i];
                break;
            }
            phi[primes[j] * i] = (primes[j] - 1) * phi[i];
        }
    }
}
```

## 1.5扩展欧几里得算法

对于 a * x + b * y = gcd(a, b), 求x和y

对于求解更一般的方程 a * x + b * y = c, 设 g = gcd(a, b) , 则其有解当且仅当 g | c

```cpp
LL exgcd(LL a, LL b, LL &x, LL &y)
{
    if (b == 0)
    {
        x = 1, y = 0;
        return a;
    }
    LL g = exgcd(b, a % b, y, x);
    y -= a / b * x;
    return g;
}
```

## 1.7中国剩余定理

### 1.7.1中国剩余定理

```cpp
int n;
int A[N], B[N];

LL exgcd(LL a, LL b, LL &x, LL &y)
{
    if(b == 0)
    {
        x = 1, y = 0;
        return a;
    }

    int g = exgcd(b, a % b, y, x);
    y -= a / b * x;

    return g;
}

int main()
{
    scanf("%d", &n);

    LL M = 1;
    for(int i = 0; i < n; i ++)
    {
        scanf("%d%d", &A[i], &B[i]);
        M *= A[i];
    }

    LL res = 0;
    for(int i = 0; i < n; i ++)
    {
        LL Mi = M / A[i];
        LL ti, x;
        exgcd(Mi, A[i], ti, x);
        res += B[i] * Mi * ti;
    }

    cout << (res % M + M) % M << endl;

    return 0;
}
```

### 1.7.2扩展中国剩余定理

即 不需要m1 m2 ... mn互质

思路，每次将两个式子化成一个式子

```cpp
LL exgcd(LL a, LL b, LL &x, LL &y)
{
    if(b == 0)
    {
        x = 1, y = 0;
        return a;
    }

    LL d = exgcd(b, a % b, y, x);
    y -= a / b * x;

    return d;
}

int main()
{
    int n;
    cin >> n;

    LL x = 0, a1, m1;
    cin >> a1 >> m1;
    for(int i = 0; i < n - 1; i ++)
    {
        LL a2, m2;
        cin >> a2 >> m2;

        LL k1, k2;
        LL d = exgcd(a1, a2, k1, k2);

        if((m2 - m1) % d != 0)
        {
            x = -1;
            break;
        }

        k1 *= (m2 - m1) / d; // 获得特解

        LL t = a2 / d;
        k1 = (k1 % t + t) % t; // 在通解中找到最小的解

        x = k1 * a1 + m1; // 目前两个方程合并完后的x的特解

        LL a = a1 / d * a2; // a1 a2的最小公倍数
        m1 = k1 * a1 + m1;
        a1 = a;
    }

    if(x != -1) x = (m1 % a1 + a1) % a1; // 最小正整数解

    cout << x << endl;

    return 0;
}
```

## 1.8逆元

### 1.8.1定义

逆元：若$a * x \equiv (mod\:b)$,且 a 与 b 互质，即称 x 为 a 的逆元，记为$a^{-1}$.

eg. 对于$\frac{x}{y} (mod\:p)$，我们就可以求出y在 mod p 下的逆元，然后乘上x，就是这个分数的值了

### 1.8.2快速幂、费马小定理

若p为质数，a为正整数，且a、p互质，则有$a^{p - 1} \equiv(mod\:p)$,则$a^{p - 2}$即为a的逆元。

```cpp
int qmi(int a, int k, int mod)
{
	int res = 1;
    while(k)
    {
        if(k & 1) res = (LL)res * a % mod;
        a = (LL)a * a % mod;
        k >>= 1;
    }
    
    return res;
}

int get_inv(int x, int p)
{
    if(x % p == 0) return ?; //没有逆元
    return qmi(x, p - 2, p);
}
```

### 1.8.4线性算法

求 1∼n 中所有整数在模 p 意义下的乘法逆元

推导：（参考洛谷zjp_shadow的题解）

首先有，$1^{-1} \equiv 1 (mod\;p)$，

然后设$p = k * i + r$,(1 < r < i < p)，也就是k是$\frac{p}{i}$的商，r是余数。

再将这个式子放到(mod p)的意义下，就会得到，

$k * i + r \equiv 0(mod\;p)$

然后两边乘上$i^{-1}$和$r^{-1}$(这里指的是对于质数p的逆元)就可以得到，

$k * r^{-1} + i^{-1} \equiv 0(mod\;p)$

$i^{-1}\equiv -k * r^{-1}(mod\:p)$

$i^{-1} \equiv \lfloor \frac{p}{i} \rfloor * (p\;mod\;i)^{-1} (mod\;p)$

```cpp
inv[1] = 1;
for(int i = 1; i < p; i ++)
{
    inv[i] = (p - p / i) * inv[p % i] % p;
}
```
